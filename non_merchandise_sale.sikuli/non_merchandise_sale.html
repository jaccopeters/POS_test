from sikuli import *
import general_functions
import general_buttons
import test_cases
import test_settings
import time



def non_merchandise_sale():
    general_functions.errorhandling("ABORT")
    l_general_test_settings = test_settings.general_test_settings()
    locale = l_general_test_settings[8]
    l_test_settings = test_settings.non_merchandise_sale_test_settings(locale)

    print('length: ' + str(len(l_test_settings)))
    x = 0
    while x < len(l_test_settings):
        print('x: ' + str(x))
        y = l_test_settings[x]
        print('y: ' + str(y))

        # click NON MERCHANDISE ITEMS button
        general_buttons.click_element('main_page_items_GV_menu', 70, 0, 'NON MERCHANDISE ITEMS btn')

        # choose non merchandise type
        general_buttons.click_element('secondary_pages_non_merchandise_sale_item_btns', y[3],
                                      y[4], 'one off the non merchandise items btn')

        # enter price
        type(str(y[5]))

        # confirm
        type(Key.ENTER)
        x = x + 1

    # click TOTAL button
    general_buttons.click_element('secondary_pages_key_pad', 100, 100, 'TOTAL btn')

    general_functions.four_eyes(l_general_test_settings[4], l_general_test_settings[5])

    general_functions.pay_with_off_line_card()
