from sikuli import *
import general_functions
import general_buttons
import test_cases
import test_settings



def pay_in():
    general_functions.errorhandling("ABORT")
    l_general_test_settings = test_settings.general_test_settings()

    general_buttons.click_element('main_page_footer_menu', 70, 0, 'OTHER btn')
    general_buttons.click_element('main_page_other_menu', -200, 70, 'PAY IN btn')

    general_functions.four_eyes( l_general_test_settings[4],  l_general_test_settings[5])

    # test_cases.errors_and_warnings_tc_0010("are the validations for the 4-eyes screen OK?")

# select pay in reason 
    type(Key.ENTER)
# pay out amount
    type("99.00")
# confirm amount
    type(Key.ENTER)
# click CASH button
# "basket_page_key_pad_and_footer_menu_en_GB.png"
    general_buttons.click_element('basket_page_key_pad_and_footer_menu', 120, -180, 'CASH btn')
# confirm pay in
    type(Key.ENTER)
